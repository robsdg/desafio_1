def menu():
    return input("""
[d] Depositar
[s] Sacar
[e] Extrato
[nu] Novo Usuário
[nc] Nova Conta
[q] Sair

=> """)

def depositar(saldo, valor, extrato, /):
    if valor > 0:
        saldo += valor
        extrato += f"Depósito\tR$ {valor:.2f}\n"
        print("\tDepósito realizado com sucesso!")
    else:
        print("\tOperação falhou! O valor informado é inválido.")
    return saldo, extrato

def sacar(*, saldo, valor, extrato, limite, numero_saques, limite_saques):
    excedeu_saldo = valor > saldo
    excedeu_limite = valor > limite
    excedeu_saques = numero_saques >= limite_saques

    if excedeu_saldo:
        print("\tOperação falhou! Você não tem saldo suficiente.")
    elif excedeu_limite:
        print("\tOperação falhou! O valor do saque excede o limite.")
    elif excedeu_saques:
        print("\tOperação falhou! Número máximo de saques excedido.")
    elif valor > 0:
        saldo -= valor
        extrato += f"Saque\t\tR$ {valor:.2f}\n"
        numero_saques += 1
        print("\tSaque realizado com sucesso!")
    else:
        print("\tOperação falhou! O valor informado é inválido.")
    return saldo, extrato, numero_saques

def exibir_extrato(*, saldo, extrato):
    print("\n================ EXTRATO ================")
    print("Não foram realizadas movimentações." if not extrato else extrato)
    print(f"\nSaldo\t\tR$ {saldo:.2f}")
    print("==========================================")

def criar_usuario(*, usuarios):
    cpf = input("Informe o CPF (somente números): ")
    usuario = filtrar_usuario(cpf=cpf, usuarios=usuarios)

    if usuario:
        print("\tJá existe um usuário com esse CPF!")
        return

    nome = input("Informe o nome completo: ")
    data_nascimento = input("Informe a data de nascimento (dd-mm-aaaa): ")
    endereco = input("Informe o endereço (logradouro, nº - bairro - cidade/UF): ")

    usuarios.append({
        "nome": nome,
        "cpf": cpf,
        "data_nascimento": data_nascimento,
        "endereco": endereco
    })
    print("\tUsuário criado com sucesso!")

def filtrar_usuario(*, cpf, usuarios):
    for usuario in usuarios:
        if usuario["cpf"] == cpf:
            return usuario
    return None

def criar_conta(*, agencia, numero_conta, usuarios, contas):
    cpf = input("Informe o CPF do usuário: ")
    usuario = filtrar_usuario(cpf=cpf, usuarios=usuarios)

    if usuario:
        contas.append({
            "agencia": agencia,
            "numero_conta": numero_conta,
            "usuario": usuario
        })
        print("\tConta criada com sucesso!")
        print(f"\tAgência:\t{agencia}")
        print(f"\tNúmero da Conta:\t{numero_conta}")
        print(f"\tTitular:\t{usuario['nome']}")
    else:
        print("\tUsuário não encontrado! Por favor, crie um usuário antes.")

def main():
    saldo = 0
    limite = 500
    extrato = ""
    numero_saques = 0
    LIMITE_SAQUES = 3
    AGENCIA = "0001"
    usuarios = []
    contas = []

    while True:
        opcao = menu()

        if opcao == "d":
            valor = float(input("Informe o valor do depósito: "))
            saldo, extrato = depositar(saldo, valor, extrato)  # Posicional only

        elif opcao == "s":
            valor = float(input("Informe o valor do saque: "))
            saldo, extrato, numero_saques = sacar(
                saldo=saldo,
                valor=valor,
                extrato=extrato,
                limite=limite,
                numero_saques=numero_saques,
                limite_saques=LIMITE_SAQUES
            )  # Nomeados obrigatórios

        elif opcao == "e":
            exibir_extrato(saldo=saldo, extrato=extrato)

        elif opcao == "nu":
            criar_usuario(usuarios=usuarios)

        elif opcao == "nc":
            numero_conta = len(contas) + 1
            criar_conta(
                agencia=AGENCIA,
                numero_conta=numero_conta,
                usuarios=usuarios,
                contas=contas
            )

        elif opcao == "q":
            print("\n\tObrigado por usar nosso sistema bancário. Até logo!")
            break

        else:
            print("\tOperação inválida, por favor selecione novamente a operação desejada.")

main()
